unsigned srl(unsigned x, int k) 
    /*Perform shift arithmetically*/
{ 
	unsigned xsra = (int)x >> k; 
	int w = sizeof(int) << 3;  //Complete *8 operations
	int s1 = (int)-1 << (w - k); //There is a small drawback here, refer to the standard answer
	int s2 = ~s1;
	return xsra & s2; 
}

//如果第一位是1，与上个相同；如果是0，则刚好相反 
int sra(int x, int k)
{
		int xsrl = (unsigned)x >> k; 
		int w = sizeof(int) << 3;
		int s1 = (int)-1 << (w - k);  //There is a small drawback here, refer to the standard answer
    	int s2 = 1 << (w - 1); 
		int s3 = s1&(! (x & s2) - 1) ;
		return xsrl | s3; 
}

#include<stdio.h>

int xbyte(packet_t word, int byteum)
{
return (word>>(byteum<<3))&0xFF; 
}

/*   1.对byteum左移3位，即让指定的字节*8得到位数；
     2.让word右移8*byte位 ，使得指定的字节处于最低有效字节；
	 3.&0xFF,只保留8位（最低有效字节）；
	得到的结果是 ：指定的字节右移到编号0 ，但是 1~3均为0，并不是符号拓展。 
*/

//B.
int ybyte(packet_t word, int byteum)
{
	int s1 = (3-byteum)<<3;
	int s2 = (word<<s1);
	return(s2>>24);
}
// 从最高位再右移24为，属于有符号数的右移，满足符号拓展。

#include<stdio.h>
#include<stdlib.h> 
int saturating_add(int x, int y)
{
	 //Solution: 将两个正数x,y相加，正溢出符号位变负，返回Tmax; 将两个负数x,y相加，符号位变正，负溢出返回Tmin; 
	int result = x+y;
	//用算术右移来保留符号位，从而判断正负溢出。
	int bias = ( sizeof(int)<<3 )-1 ; //记录算术右移的步数；
	int mark1 = (~(x^y))>>bias;  //x,y判断符号位
	int mark2 = (result^y)>>bias; // 判断reuslt的符号位是否溢出
	int mark = mark1 & mark2;  // 判断溢出条件
	int x_sign = x>>bias;
	
	int judge =  result & (~(mark));  //溢出的判定结果 
	int pos = (1<<bias) & x_sign;
	int neg = mark & ((-1) ^ (1 << bias)) & ~mark;
	
	int ans = judge + pos + neg;
}

int main()

	int a = 0, b = 0;
	scanf("%d%d", &a, &b);
	printf("%d\n", saturating_add(a, b));
	return 0;
}


/*参考答案
/* 饱和到TMin/TMax的加法 */
 int saturating_add(int x, int y) {
 int sum = x + y;
 int wm1 = (sizeof(int)<<3)-1; // 31

  /* 如果符号位为0，则对应的掩码为全0；如果符号位为1，则对应的掩码为全1 */

  int xneg_mask = (x >> wm1);

  int yneg_mask = (y >> wm1);

  int sneg_mask = (sum >> wm1);

  /* 正溢出的判断条件：x和y都是正数，sum为负数 */
 int pos_over_mask =  ~xneg_mask &  ~yneg_mask & sneg_mask;
  /* 负溢出的判断条件：x和y都是负数，sum为正数 */
 int neg_over_mask = xneg_mask & yneg_mask &  ~sneg_mask;
     
  /* 有正溢出或者负溢出就判定为有溢出 */
    // substitue '&' to 'if' in judgement oprations
int over_mask = pos_over_mask | neg_over_mask;

  /* 在sum、INT_MAX和INT_MIN中选择：

        无溢出选择sum，有正溢出选择INT_MAX，有负溢出选择INT_MIN */
 int result = (~over_mask & sum) |

               (pos_over_mask & INT_MAX) |

               (neg_over_mask & INT_MIN);

return result;
 }
